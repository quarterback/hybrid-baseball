    • {% for p in range(1, page_count + 1) %} {% if p == 1 or p == page_count or (p >= page - 2 and p <= page + 2) %}
  • {{ p }}
    • {% elif p == page - 3 or p == page + 3 %}
  • …
    • {% endif %} {% endfor %}
{% endif %} {% endmacro %}
